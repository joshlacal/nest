14pub fn create_signed_jwt<C>(
15    key: SigningKey<C>,
16    header: Header,
17    claims: Claims,
18) -> serde_json::Result<String>
19where
20    C: PrimeCurve + CurveArithmetic + DigestPrimitive,
21    Scalar<C>: Invert<Output = CtOption<Scalar<C>>> + SignPrimitive<C>,
22    SignatureSize<C>: ArrayLength<u8>,
23{
24    let header = URL_SAFE_NO_PAD.encode(serde_json::to_string(&header)?);
25    let payload = URL_SAFE_NO_PAD.encode(serde_json::to_string(&claims)?);
26    let signature: Signature<_> = key.sign(format!("{header}.{payload}").as_bytes());
27    Ok(format!("{header}.{payload}.{}", URL_SAFE_NO_PAD.encode(signature.to_bytes())))
28}
